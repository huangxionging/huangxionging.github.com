- (void)setEditing:(BOOL)editing animated:(BOOL)animated; 
// Updates the appearance of the Edit|Done button item as necessary. Clients who override it must call super first. 中文翻译：如果编辑或完成按钮项目有这个需要，系统将会更新视图。客户重载这个方法，必须首先调用父类的这个方法

- (void) setEditing:(BOOL)editing animated:(BOOL)animated
{
    [super setEditing:editing animated:animated];
    [self.tabView setEditing: editing animated: animated];
    if (editing)
    {
        self.textField.hidden = NO;
    }
    else
    {
        self.textField.hidden = YES;
    }
}
