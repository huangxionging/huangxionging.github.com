CGContextRef 是绘图环境, 绘图设备都可以.

在 iOS 中大多会使用 UIKit 中 UIGraphics.h 中的 UIGraphicsGetCurrentContext 这个函数来获得 CGContext

CGContextRotateCTM(CGContextRef c, CGFloat angle) // 该函数的功能是旋转整个 CGContext, angle 是角度 0~2π.

CGContextScaleCTM(CGContextRef c, CGFloat sx, CGFloat sy) // 缩放 sx, sy 是横纵缩放比例

CGContextTranslateCTM(CGContextRef c, CGFloat tx, CGFloat ty) // 平移 tx ty 是平移坐标

CGAffineTransform CGContextGetCTM(CGContextRef c) // 获取当前绘图环境的仿射变换

CGContextSetLineWidth(CGContextRef c, CGFloat width) // 设置线宽

CGContextSetLineCap(CGContextRef c, CGLineCap cap) // 设置线帽 CGLineCap 为线帽

CGContextSetLineJoin(CGContextRef c, CGLineJoin join) // 连接方式 CGLineJoin 是连接方式

CGContextSetMiterLimit(CGContextRef c, CGFloat limit) // 设置转角限量

CGContextSetLineDash(CGContextRef c, CGFloat phase, const CGFloat lengths[], size_t count) // 添加虚线 lengths[]是虚线数组, 从第一个数字开始, 依次按实虚实虚的秩序绘制 phase, 是绘制第一条实线省去的长度 count 是 lengths 数组的个数

CGContextSetFlatness(CGContextRef c, CGFloat flatness) // 设置弯曲的路径中的图形上下文的准确性。路径平滑的程度

CGContextSetAlpha(CGContextRef c, CGFloat alpha) // 设置 alpha 值, 可见度

CGContextSetBlendMode(CGContextRef context, CGBlendMode mode) // 设置混合模式

CGContextBeginPath(CGContextRef c) // 开始绘制新的路径, 旧路径被抛弃

CGContextMoveToPoint(CGContextRef c, CGFloat x, CGFloat y) // 移动到原点

CGContextAddLineToPoint(CGContextRef c, CGFloat x, CGFloat y) // 从原点到目标点绘制直线

CGContextAddCurveToPoint(CGContextRef c, CGFloat cp1x, CGFloat cp1y, CGFloat cp2x, CGFloat cp2y, CGFloat x, CGFloat y) // 绘制曲线, 使用贝塞尔方法绘制, 三次贝塞尔曲线

CGContextAddQuadCurveToPoint(CGContextRef c, CGFloat cpx, CGFloat cpy, CGFloat x, CGFloat y) // 绘制二次曲线, 二次贝塞尔曲线

CGContextClosePath(CGContextRef c) // 关闭当前的绘制路径

CGContextAddRect(CGContextRef c, CGRect rect) // 添加矩形

CGContextAddRects(CGContextRef c, const CGRect rects[], size_t count) // 批量添加矩形

CGContextAddLines(CGContextRef c, const CGPoint points[], size_t count) // 批量添加线条

CGContextAddEllipseInRect(CGContextRef context, CGRect rect) // 绘制椭圆

CGContextAddArc(CGContextRef c, CGFloat x, CGFloat y, CGFloat radius, CGFloat startAngle, CGFloat endAngle, int clockwise) // 绘制扇形, x, y 是中心, radius 是半径, startAngle 是开始角度, endAngle 是结束角度, clockwise 是顺时针, clockwise 值为1表示顺时针,0表示不是.

CGContextAddArcToPoint(CGContextRef c, CGFloat x1, CGFloat y1, CGFloat x2, CGFloat y2, CGFloat radius) // 两个点之间加上半径确立扇形

void CGContextAddPath(CGContextRef context, CGPathRef path) // 添加路径

CG_EXTERN void CGContextReplacePathWithStrokedPath(CGContextRef c) // 功能是使用新的Path 来替代之前的 path 通常 CGContextClip 后面使用

CG_EXTERN bool CGContextIsPathEmpty(CGContextRef context) // 判断当前 context 是否有 path 路径

CG_EXTERN CGPoint CGContextGetPathCurrentPoint(CGContextRef context) // 获取当前路径所在的点, 比如绘制了一条线之后, 路径当前的位置

CG_EXTERN CGRect CGContextGetPathBoundingBox(CGContextRef context) // 获取能够包含路径上所有点的最小矩形

CG_EXTERN CGPathRef CGContextCopyPath(CGContextRef context) // 复制当前 context 上的 path

/** Shadow support. **/

/* Set the shadow parameters in `context'. `offset' specifies a translation

/** Transparency layer support. **/

/* Begin a transparency layer in `context'. All subsequent drawing
   operations until a corresponding `CGContextEndTransparencyLayer' are
   composited into a fully transparent backdrop (which is treated as a
   separate destination buffer from the context). After the transparency
   layer is ended, the result is composited into the context using the
   global alpha and shadow state of the context. This operation respects the
   clipping region of the context. After a call to this function, all of the
   parameters in the graphics state remain unchanged with the exception of
   the following:
     - The global alpha is set to 1.
     - The shadow is turned off.
     - The blend mode is set to `kCGBlendModeNormal'.
   Ending the transparency layer restores these parameters to the values
   they had before `CGContextBeginTransparencyLayer' was called.
   Transparency layers may be nested. */

CG_EXTERN void CGContextBeginTransparencyLayer(CGContextRef context,
  CFDictionaryRef auxiliaryInfo)
  CG_AVAILABLE_STARTING(__MAC_10_3, __IPHONE_2_0);

  `CGContextBeginTransparencyLayer' except that the content of the
  transparency layer will be bounded by `rect' (specified in user space). */

CG_EXTERN void CGContextBeginTransparencyLayerWithRect(CGContextRef context,
  CGRect rect, CFDictionaryRef auxiliaryInfo)
  CG_AVAILABLE_STARTING(__MAC_10_5, __IPHONE_2_0);

CG_EXTERN void CGContextEndTransparencyLayer(CGContextRef context)
