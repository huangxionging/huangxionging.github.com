   self.scrollView.contentSize = CGSizeMake(self.view.frame.size.width * 3, self.scrollView.frame.size.height);
    self.scrollView.frame = self.view.frame;
    UIStoryboard *board = [UIStoryboard storyboardWithName: @"Main" bundle: nil];
    UIViewController *page1Controller = [board instantiateViewControllerWithIdentifier: @"page1"];
    self.page1 = page1Controller.view;
    self.page1.frame = CGRectMake(0.0f, 0.0f, 320.0f, 420.0f);
    UIViewController *page2Controller = [board instantiateViewControllerWithIdentifier: @"page2"];
    self.page2 = page2Controller.view;
    self.page2.frame = CGRectMake(320.0f, 0.0f, 320.0f, 420.0f);
    UIViewController *page3Controller = [board instantiateViewControllerWithIdentifier: @"page3"];
    self.page3 = page3Controller.view;
    self.page3.frame = CGRectMake(2 * 320.0f, 0.0f, 320.0f, 420.0f);
    self.scrollView.delegate = self;
   [self.scrollView addSubview: self.page1];
    [self.scrollView addSubview: self.page2];
    [self.scrollView addSubview: self.page3];


- (IBAction)changePage:(id)sender
{
    [UIView animateWithDuration: 0.5f animations:
     ^{
         NSInteger whichPage = self.pageControl.currentPage;
         self.scrollView.contentOffset = CGPointMake(320.0f * whichPage, 0.0f);
     }];
}

- (void) scrollViewDidScroll:(UIScrollView *)scrollView
{
    CGPoint  offSet = scrollView.contentOffset;
    self.pageControl.currentPage = offSet.x / 320.0f;
}
