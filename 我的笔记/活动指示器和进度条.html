这是用于请求的任务完成之前，活动指示器可以消除用户的心理等待时间，而进度条可以指示请求的进度。

// 点击了Upload按钮，修改活动指示器的状态
- (IBAction)startMove:(id)sender
{
    // 如果活动指示器已经在旋转，那么让它停止
    if ([self.myActivityIndicatorView isAnimating])
    {
        [self.myActivityIndicatorView stopAnimating];
    }
    else
    {
        // 如果本身是停止的，那么让它旋转
        [self.myActivityIndicatorView startAnimating];
    }
}


// 点击Download按钮
- (IBAction)downloadProgress:(id)sender
{
    timer = [NSTimer scheduledTimerWithTimeInterval: 1.0 target:self selector: @selector(download) userInfo:nil repeats:YES];
}

- (void)download
{
    self.myProgressView.progress = self.myProgressView.progress + 0.1;
    if (self.myProgressView.progress == 1.0)
    {
        [timer invalidate];
        UIAlertView *alter = [[UIAlertView alloc] initWithTitle: @"download completed!" message: @"" delegate:nil cancelButtonTitle: @"OK" otherButtonTitles:nil, nil];
        [alter show];
        self.myProgressView.progress = 0;
    }
}
