CoreGraphics 框架是 iOS 和 OS X 通用的 c 语言接口. 是一套图形接口, 用来处理各种图形处理.

#ifndef COREGRAPHICS_H_
#define COREGRAPHICS_H_

#include <CoreGraphics/CGBase.h>

// represents a matrix used for affine transformations
#include <CoreGraphics/CGAffineTransform.h>

// Describes an opaque type for managing drawing operations in a bitmap.
#include <CoreGraphics/CGBitmapContext.h>

// 颜色 Describes a representation of color and opacity (alpha value).

#include <CoreGraphics/CGColor.h>

// 颜色空间 Describes an opaque type for specifying color space information.
#include <CoreGraphics/CGColorSpace.h>

// 上下文 Describes a Quartz 2D drawing destination.
#include <CoreGraphics/CGContext.h>

// 数据消费 Describes an opaque data type for writing image or PDF data.
#include <CoreGraphics/CGDataConsumer.h>

// 数据提供 Describes an opaque data type for reading image data.
#include <CoreGraphics/CGDataProvider.h>

// 错误 
#include <CoreGraphics/CGError.h>

// 字体 Describes a representation of a font.
#include <CoreGraphics/CGFont.h>

// 函数 Describes a facility for defining and using callback functions.
#include <CoreGraphics/CGFunction.h>

// 几何
#include <CoreGraphics/CGGeometry.h>

// 斜度 Describes the functions that create and operate on Quartz gradient objects.
#include <CoreGraphics/CGGradient.h>

// 图像 Describes a representation of an image.
#include <CoreGraphics/CGImage.h>

// 层 Describes the opaque type for creating and drawing to a Quartz layer
#include <CoreGraphics/CGLayer.h>

// PDF 数组 Describes an array structure in a PDF document
#include <CoreGraphics/CGPDFArray.h>

// PDF 内容流 Describes the appearance of a PDF page.
#include <CoreGraphics/CGPDFContentStream.h>

// PDF上下文 Describes a Quartz PDF drawing destination.
#include <CoreGraphics/CGPDFContext.h>

// PDF 字典 Describes a dictionary structure in a PDF document
#include <CoreGraphics/CGPDFDictionary.h>

// PDF 文档 Describes a document containing PDF data
#include <CoreGraphics/CGPDFDocument.h>

// PDF 对象 Describes a PDF object in a PDF document
#include <CoreGraphics/CGPDFObject.h>

// PDF 操作表 Describes a collection of PDF operators and associated callback functions
#include <CoreGraphics/CGPDFOperatorTable.h>

// PDF 页面 Describes a page in a PDF document
#include <CoreGraphics/CGPDFPage.h>

// PDF 查看器 Describes an opaque type for parsing a PDF content stream
#include <CoreGraphics/CGPDFScanner.h>

// PDF 流 Describes a PDF data stream, which typically represents an image or a page description
#include <CoreGraphics/CGPDFStream.h>

// PDF 字符串 Describes a character string in a PDF document
#include <CoreGraphics/CGPDFString.h>

// 路径 Describes a vector-based path consisting of line and curve segments.
#include <CoreGraphics/CGPath.h>

// 模式 Describes a representation of a fill pattern
#include <CoreGraphics/CGPattern.h>

// Describes a gradient fill pattern.
#include <CoreGraphics/CGShading.h>


#endif /* COREGRAPHICS_H_ */
